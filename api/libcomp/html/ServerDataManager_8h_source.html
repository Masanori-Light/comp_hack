408  template <class T>
409  bool LoadObjectsFromFile(gsl::not_null<DataStore*> pDataStore,
410  const libcomp::String& filePath,
411  DefinitionManager* definitionManager = nullptr)
412  {
413  tinyxml2::XMLDocument objsDoc;
414 
415  std::vector<char> data = pDataStore->ReadFile(filePath);
416 
417  if(data.empty())
418  {
419  LogServerDataManagerWarning([&]()
420  {
421  return String("File does not exist or is empty: %1\n")
422  .Arg(filePath);
423  });
424 
425  return true;
426  }
427 
428  if(tinyxml2::XML_SUCCESS !=
429  objsDoc.Parse(&data[0], data.size()))
430  {
431  return false;
432  }
433 
434  const tinyxml2::XMLElement *rootNode = objsDoc.RootElement();
435  const tinyxml2::XMLElement *objNode = rootNode->FirstChildElement("object");
436 
437  while(nullptr != objNode)
438  {
439  if(!LoadObject<T>(objsDoc, objNode, definitionManager))
440  {
441  LogServerDataManagerError([&]()
442  {
443  return String("Failed to load XML file: %1\n")
444  .Arg(filePath);
445  });
446 
447  return false;
448  }
449 
450  objNode = objNode->NextSiblingElement("object");
451  }
452 
453  LogServerDataManagerInfo([&]()
454  {
455  return String("Loaded XML file: %1\n").Arg(filePath);
456  });
457 
458  return true;
459  }
